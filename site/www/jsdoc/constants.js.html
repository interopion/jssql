RE_ISO8601 = new RegExp([
	"^",
	"(",                                 // \1  - date
		"\\s*",
		"(\\d{4})",                      // \2  - year
		"-",
		"(0[1-9]|1[0-2]|[1-9])",         // \3  - month
		"-",
		"([12]\\d|0[1-9]|3[01]|[1-9])",  // \4  - day
	")?",
	"(",                                 // \5  - time
		"[tT\\s]?",                           
		"([01]\\d|2[0-3])",              // \6  - hours
		"\\:",
		"([0-5]\\d|\\d)",                // \7  - minutes
		"(\\:",                          // \8  - seconds and milliseconds
			"([0-5]\\d|\\d)",            // \9  - seconds
			"([\\.,](\\d+))?",           // \11 - milliseconds
		")?",
		"\\s*(",                         // \12 - timezone
			"[zZ]|",
			"(",
				"([\\+\\-])",            // \14 - timezone direction
				"([01]\\d|2[0-3]|\\d)",  // \15 - timezone hours
				"(\\:?([0-5]\\d|\\d))?", // \17 - timezone minutes
			")",
		")?", 
	")?",
	"\\s*$"
].join("")),
